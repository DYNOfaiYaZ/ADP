#include <iostream>
using namespace std;
int main()
{
    int a,b,c;
    int u,v;
    int y;
    int i,j,k;
    cin>>a>>b>>c>>u>>v>>i>>j>>k;
    int x=a + b * 5 / 4 +c % 3*5;
    u>v?y=u:y=v;
    int z = ++i&&++j&&++k;
    cout<<"x = "<<x;
    cout<<"y = "<<y;
    cout<<"z = "<<z;
}
