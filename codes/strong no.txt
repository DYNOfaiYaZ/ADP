#include <iostream>
using namespace std;

int fact(int n)
{
    int fact = 1;
    
    for(int i = 1; i <= n; i++)
            fact = fact * i;
    
    return fact;
}

int Strgn(int n)
{
    int digit;
    int sum = 0;
    int temp = n;
    
    while(temp!=0)
    {
        digit = temp % 10;
        sum += fact(digit);
        temp /= 10;
    }
    return sum == n;
}
int main ()
{
    int n;
    cin>>n;
    
    if(Strgn(n))
    {
        cout << n << " is Strong Number";
    }
    else
    {
        cout << n<< " is Not Strong Number";
    }
    
}
