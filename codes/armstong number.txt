#include <iostream>
#include <math.h>
using namespace std;

int main() {
    int n,r;
    int sum= 0;
    cin >> n;
    int temp=n;
    int temp2=n;
    int count = 0;
  
    while (n != 0) 
    {
        n = n / 10;
        ++count;
    }

    while (temp != 0) 
    {
        r = temp % 10;
        sum += pow(r,count);
        temp/= 10;
    }

    if (sum == temp2)
    {
        cout << temp2 << " is an Armstrong number.";
    }
    else
    {
        cout << temp2 << " is not an Armstrong number.";
    }
    
}
