#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    
    if(n%2==0)
    {
        cout<<"Enter ODD number of rows";
    }
    else
    {
        for(int i=1;i<=n;i++)
        {
            int col=(i<=n / 2 + 1) ? i : n - i + 1;
        
            for(int j=1;j<=col;j++)
            {
                cout<<i*j<<" ";
            }
            cout<<endl;
        }
    }
    
  
}
