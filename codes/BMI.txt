#include <iostream> 
#include <math.h> 
using namespace std; 
int main(){
    
  float h,w;
  cin>>h>>w;
  h=h/100;
  float bmi=w/h*h;
 
    if (bmi < 18.5) {
        cout << "underweight"; 
    }
    else if (bmi >= 18.5 && bmi < 24.9) {
        cout << "Healthy"; 
    }
    else if (bmi >= 24.9 && bmi < 30){ 
        cout << "overweight"; 
    }
    else if (bmi >= 30) {
        cout << "Suffering from Obesity"; 
    }
    return 0; 
}
