#include <iostream>
using namespace std;
int main(){
    int n;
    cin>>n;
    int sum = 0;
    
    for(int d = 1; d < n; d++){
        if(n%d == 0)
            sum = sum+d;
    }
    
    if(sum == n)                                   |               
        cout << n << " is a perfect number";       |    
    else                                           |     sum==n?cout<<n<<"perfect":cout<<n<<"not perfect";
        cout << n << " is not a perfect number";   |
}
