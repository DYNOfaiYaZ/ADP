#include <iostream>
using namespace std;
int main()

{
    int a,i,x=0,y=0,z=0;
    cin>>a;
    for(int s=1;s<=a;s++)
    {
        cin>>i;
        if (i>=0 && i<=5)
        {
            x=x + 1;
        }
        else if (i>=6 && i<=17)
        {
            y=y + 1;
        }
        else if (i>=18)
        {
            z=z + 1;
        }
    }
    cout<<"Baby: "<<x<<endl;
    cout<<"Attending school: "<<y<<endl;
    cout<<"Adult: "<<z<<endl;
    
}
