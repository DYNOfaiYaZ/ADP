#include <iostream>
using namespace std;
int main()
{
    int n,row,col;
    cin>>n;
    for(row=1;row<=n;row++)
    {
        for(col=1;col<=row;col++)
        {
            if(col==1 || col==row || row==n) 
            {
                cout<<"* ";
            }
            else
            {
                cout<<"  ";
            }
        }
        
        cout<<endl;
    }
}

#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n-i;j++)
        {
            cout<<" ";
        }
        for(int j=1;j<=(2*i - 1);j++))
        {
            if(j==1 || j==(2*i - 1) || i==n)
            {
                cout<<"*";
            }
            else
            {
                cout<<" ";
            }
        }
        cout<<endl;
    }
}
